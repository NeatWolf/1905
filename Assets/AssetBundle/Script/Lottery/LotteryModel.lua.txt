LotteryModel = {}

-- 定义概率
LotteryModel.Normal = {}
LotteryModel.Plus = {}
LotteryModel.Normal.roleProbability = 0.2 -- 角色概率
LotteryModel.Normal.L3ItemProbability = 0.2 -- L3道具
LotteryModel.Normal.L2ItemProbability = 0.5 -- L2道具
LotteryModel.Plus.roleProbability = 0.2 -- 角色概率
LotteryModel.Plus.L3ItemProbability = 0.2 -- L3道具
LotteryModel.Plus.L2ItemProbability = 0.5 -- L2道具

-- 扣资源
function LotteryModel:DeductAsset(type)
    -- 扣除资源 =============================== 定义抽卡价格
    if type == "Normal" then
        PlayerDataModel.PlayerData.Money = PlayerDataModel.PlayerData.Money - 5000
    elseif type == "Plus" then
        PlayerDataModel.CompletePlayerData.Diamond = PlayerDataModel.CompletePlayerData.Diamond - 100
    else
        print("暂无此卡池类型")
        return
    end
end
